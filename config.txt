# Define blocked IPs and suspicious keywords
BLOCKED_IPS = {"192.168.1.50", "10.0.0.99"}
SUSPICIOUS_KEYWORDS = ["malware", "attack", "phishing", "exploit"]

# Log file
LOG_FILE = "firewall_log.txt"

def log_event(event):
    """Log firewall events to a file with timestamp."""
    with open(LOG_FILE, "a") as f:
        f.write(f"[{time.strftime('%Y-%m-%d %H:%M:%S')}] {event}\n")

def check_packet(ip, data):
    """Check incoming packet for suspicious activity."""
    if ip in BLOCKED_IPS:
        log_event(f"Blocked packet from {ip}")
        return False
    for keyword in SUSPICIOUS_KEYWORDS:
        if keyword in data.lower():
            log_event(f"Suspicious content detected from {ip}: {keyword}")
            return False
    return True

def handle_client(client_socket, client_address):
    """Handle incoming client connection."""
    ip = client_address[0]
    try:
        while True:
            data = client_socket.recv(1024).decode("utf-8")
            if not data:
                break
            if check_packet(ip, data):
                log_event(f"Allowed traffic from {ip}: {data}")
                client_socket.send("Packet allowed\n".encode("utf-8"))
            else:
                client_socket.send("Packet blocked\n".encode("utf-8"))
    except Exception as e:
        log_event(f"Error handling client {ip}: {e}")
    finally:
        client_socket.close()

def start_firewall(host="0.0.0.0", port=9999):
    """Start the firewall server."""
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind((host, port))
    server.listen(5)
    log_event(f"Firewall started on {host}:{port}")
    print(f"Firewall running on {host}:{port}...")

    while True:
        client_socket, client_address = server.accept()
        log_event(f"Connection attempt from {client_address[0]}")
        client_handler = threading.Thread(
            target=handle_client, args=(client_socket, client_address)
        )
        client_handler.start()

if __name__ == "__main__":
    start_firewall()

"""
Cybersecurity Log Analyzer
Author: Example
Purpose: Parse system logs, detect suspicious activity, and generate a report.
"""

import re
import os
import datetime
from collections import defaultdict

# Step 1: Define suspicious patterns
SUSPICIOUS_PATTERNS = {
    "failed_login": re.compile(r"Failed password for .* from (\d+\.\d+\.\d+\.\d+)"),
    "port_scan": re.compile(r"Connection attempt from (\d+\.\d+\.\d+\.\d+) on port (\d+)"),
    "sql_injection": re.compile(r"(?:SELECT|UNION|DROP|INSERT).*", re.IGNORECASE),
}

# Step 2: Load log file
LOG_FILE = "system.log"

if not os.path.exists(LOG_FILE):
    print(f"Log file {LOG_FILE} not found. Please provide a valid log file.")
    exit(1)

# Step 3: Parse logs
alerts = defaultdict(list)

with open(LOG_FILE, "r") as f:
    for line in f:
        for category, pattern in SUSPICIOUS_PATTERNS.items():
            match = pattern.search(line)
            if match:
                alerts[category].append((line.strip(), match.groups()))

# Step 4: Generate report
timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
report_file = "security_report.txt"

with open(report_file, "w") as report:
    report.write(f"Cybersecurity Log Analysis Report\n")
    report.write(f"Generated: {timestamp}\n\n")

    if not alerts:
        report.write("No suspicious activity detected.\n")
    else:
        for category, entries in alerts.items():
            report.write(f"=== {category.upper()} ===\n")
            for entry, details in entries:
                report.write(f"Log: {entry}\nDetails: {details}\n\n")

print(f"Analysis complete. Report saved to {report_file}")
#!/usr/bin/env python3
"""
Cybersecurity Packet Sniffer
Purpose: Capture network packets, analyze headers, and detect suspicious traffic patterns.
"""

import socket
import struct

# Step 1: Create raw socket
# AF_PACKET works on Linux; for Windows, use AF_INET with SOCK_RAW
sniffer = socket.socket(socket.AF_PACKET, socket.SOCK_RAW, socket.ntohs(0x0003))

def ethernet_frame(data):
    dest_mac, src_mac, proto = struct.unpack('!6s6sH', data[:14])
    return (
        ':'.join('%02x' % b for b in dest_mac),
        ':'.join('%02x' % b for b in src_mac),
        socket.htons(proto),
        data[14:]
    )

def ipv4_packet(data):
    version_header_length = data[0]
    header_length = (version_header_length & 15) * 4
    ttl, proto, src, target = struct.unpack('!8xBB2x4s4s', data[:20])
    return (
        ttl,
        proto,
        '.'.join(map(str, src)),
        '.'.join(map(str, target)),
        data[header_length:]
    )

print("Sniffer started... Press Ctrl+C to stop.")

try:
    while True:
        raw_data, addr = sniffer.recvfrom(65535)
        dest_mac, src_mac, eth_proto, data = ethernet_frame(raw_data)
        print(f"\nEthernet Frame: {src_mac} -> {dest_mac}, Protocol: {eth_proto}")

        # IPv4 packets
        if eth_proto == 8:
            ttl, proto, src, target, data = ipv4_packet(data)
            print(f"IPv4 Packet: {src} -> {target}, TTL: {ttl}, Protocol: {proto}")

            # Detect suspicious traffic (example: unusual TTL or ICMP flood)
            if ttl < 5:
                print("⚠️ Suspicious: Very low TTL (possible crafted packet)")
            if proto == 1:  # ICMP
                print("⚠️ ICMP traffic detected (possible ping flood)")
except KeyboardInterrupt:
    print("\nSniffer stopped.")
